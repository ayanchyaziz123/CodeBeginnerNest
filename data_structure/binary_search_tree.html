Binary Tree Operations

Insertion

Description: Inserting a new node into the binary tree.

Time Complexity: O(log n) on average (balanced tree), O(n) worst-case (unbalanced tree)


class Node:
    def __init__(self, key):
        self.left = None
        self.right = None
        self.val = key

# Example insertion
root = Node(1)
root.left = Node(2)
root.right = Node(3)
root.left.left = Node(4)
root.left.right = Node(5)

Traversal

Description: Performing depth-first traversal (in-order).

Time Complexity: O(n)


def inorder_traversal(node):
    if node:
        inorder_traversal(node.left)
        print(node.val)
        inorder_traversal(node.right)

# Example traversal
inorder_traversal(root)
